#### 2.2 API响应类型统一

**问题**: 前端组件直接使用 `response.data`，但类型定义不明确

**修复方案**:

创建统一的API响应处理Hook:

```typescript
// src/hooks/useApi.ts
import { useState, useCallback } from 'react';
import { ApiResponse, ApiError } from '@/lib/tauri-api';

interface UseApiState<T> {
  data: T | null;
  loading: boolean;
  error: ApiError | null;
}

export function useApi<T>(apiCall: () => Promise<T>) {
  const [state, setState] = useState<UseApiState<T>>({
    data: null,
    loading: false,
    error: null,
  });

  const execute = useCallback(async () => {
    setState(prev => ({ ...prev, loading: true, error: null }));
    try {
      const data = await apiCall();
      setState({ data, loading: false, error: null });
      return data;
    } catch (error) {
      const apiError = error instanceof ApiError ? error : new ApiError(String(error));
      setState({ data: null, loading: false, error: apiError });
      throw apiError;
    }
  }, [apiCall]);

  return { ...state, execute };
}
```

#### 3.2 model_test.rs设计

```rust
#[cfg(test)]
mod tests {
    use super::*;
    use crate::services::config_manager::ConfigManager;
    use std::sync::Arc;
    use tempfile::TempDir;

    // 辅助函数：创建测试用的ConfigManager
    async fn create_test_config_manager() -> (Arc<ConfigManager>, TempDir) {
        let temp_dir = TempDir::new().unwrap();
        let config_path = temp_dir.path().join("test_config.yaml");

        let manager = ConfigManager::new_async(&config_path).await.unwrap();
        (Arc::new(manager), temp_dir)
    }

    #[tokio::test]
    async fn test_delete_model_success() {
        let (config_manager, _temp) = create_test_config_manager().await;

        // 先添加一个测试模型
        let test_model = create_test_model("test-model", false);
        config_manager.save_model_full(test_model.clone()).await.unwrap();

        // 删除模型
        let result = delete_model(
            "test-model".to_string(),
            State(config_manager)
        ).await;

        assert!(result.is_ok());
        let response = result.unwrap();
        assert!(response.success);
        assert_eq!(response.data, Some(true));

        // 验证模型已被删除
        let models = config_manager.get_models_full().await.unwrap();
        assert!(!models.iter().any(|m| m.id == "test-model"));
    }

    #[tokio::test]
    async fn test_delete_model_not_found() {
        let (config_manager, _temp) = create_test_config_manager().await;

        let result = delete_model(
            "non-existent-model".to_string(),
            State(config_manager)
        ).await;

        assert!(result.is_ok());
        let response = result.unwrap();
        assert!(!response.success);
        assert!(response.error.is_some());
        assert!(response.error.unwrap().contains("不存在"));
    }

    #[tokio::test]
    async fn test_delete_default_model_rejected() {
        let (config_manager, _temp) = create_test_config_manager().await;

        // 获取默认模型
        let models = config_manager.get_models_full().await.unwrap();
        let default_model = models.iter().find(|m| m.default).unwrap();

        // 尝试删除唯一的默认模型
        let result = delete_model(
            default_model.id.clone(),
            State(config_manager)
        ).await;

        assert!(result.is_ok());
        let response = result.unwrap();
        assert!(!response.success);
        assert!(response.error.unwrap().contains("不能删除唯一的默认模型"));
    }

    #[tokio::test]
    async fn test_set_default_model_success() {
        let (config_manager, _temp) = create_test_config_manager().await;

        // 添加两个模型
        let model1 = create_test_model("model-1", false);
        let model2 = create_test_model("model-2", false);
        config_manager.save_model_full(model1).await.unwrap();
        config_manager.save_model_full(model2).await.unwrap();

        // 设置model-2为默认
        let result = set_default_model(
            "model-2".to_string(),
            State(config_manager.clone())
        ).await;

        assert!(result.is_ok());
        let response = result.unwrap();
        assert!(response.success);

        // 验证model-2现在是默认
        let models = config_manager.get_models_full().await.unwrap();
        let model2_updated = models.iter().find(|m| m.id == "model-2").unwrap();
        assert!(model2_updated.default);
    }

    #[tokio::test]
    async fn test_reorder_models_success() {
        let (config_manager, _temp) = create_test_config_manager().await;

        // 添加多个模型
        for i in 0..3 {
            let model = create_test_model(&format!("model-{}", i), false);
            config_manager.save_model_full(model).await.unwrap();
        }

        // 重新排序
        let result = reorder_models(
            vec!["model-2".to_string(), "model-0".to_string(), "model-1".to_string()],
            State(config_manager.clone())
        ).await;

        assert!(result.is_ok());
        let response = result.unwrap();
        assert!(response.success);

        // 验证顺序已更新
        let models = config_manager.get_models_full().await.unwrap();
        // model-2应该有最高优先级(0)
        let model2 = models.iter().find(|m| m.id == "model-2").unwrap();
        assert_eq!(model2.priority, 0);
    }

    // 辅助函数
    fn create_test_model(id: &str, default: bool) -> ModelConfigFull {
        ModelConfigFull {
            id: id.to_string(),
            name: format!("Test Model {}", id),
            provider: "openai".to_string(),
            api_base: None,
            model: "gpt-4".to_string(),
            parameters: ModelParameters {
                temperature: 0.7,
                max_tokens: 2048,
                top_p: 1.0,
                presence_penalty: 0.0,
                frequency_penalty: 0.0,
            },
            capabilities: ModelCapabilities::default(),
            priority: 0,
            enabled: true,
            default,
        }
    }
}
```
